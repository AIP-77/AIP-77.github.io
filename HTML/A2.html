function getHourIntervalForWorkDay(timeStr, workDate) { if (!timeStr) return null; const hour = getHourFromTime(timeStr); if (hour === null) return null; if (hour < 9) { const displayStart = String(hour).padStart(2, '0'); const displayEnd = String((hour + 1) % 24).padStart(2, '0'); return { key: `${displayStart}-${displayEnd}`, display: `${displayStart}–${displayEnd} (ночь)`, shortDisplay: `${displayStart}–${displayEnd}`, // ← для графиков isNight: true, sortKey: hour + 24 }; } else { const displayStart = String(hour).padStart(2, '0'); const displayEnd = String(hour + 1).padStart(2, '0'); return { key: `${displayStart}-${displayEnd}`, display: `${displayStart}–${displayEnd}`, shortDisplay: `${displayStart}–${displayEnd}`, // ← единый формат isNight: false, sortKey: hour }; } }
